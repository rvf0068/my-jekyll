<!-- Pyodide loader (only once per page) -->
<script src="https://cdn.jsdelivr.net/pyodide/v0.29.0/full/pyodide.js"></script>
<script>
  const pyodideReady = loadPyodide();
</script>

<!-- Code cell 1 -->
<div class="py-cell">
  <textarea id="pycode1" rows="4" cols="40">2 + 2</textarea><br>
  <button id="run1">Run</button>
  <pre id="out1"></pre>
</div>

<script>
  (async () => {
    const pyodide = await pyodideReady;
    document.getElementById("run1").onclick = async () => {
      const code = document.getElementById("pycode1").value;
      try {
        const result = await pyodide.runPythonAsync(code);
        document.getElementById("out1").textContent = result;
      } catch (err) {
        document.getElementById("out1").textContent = err;
      }
    };
  })();
</script>
